/*******************************************
DDS_mode:
			1:调频
			2:AM;
			3:FM;
			4:PSK
			5:ASK
			6:调节VPP

/****************************************************************************
		       SPI	
	SPI1_SCK  	-----> PG.9   	时钟信号
	SPI1_CMD_CS    	-----> PG.11  	命令片选信号
	SPI1_DATA_CS	-----> PG.13	数据片选信号
        	SPI1_MOSI 	-----> PG.15   	主设备数据输出，从设备数据输入

/****************************************************************************
AM调制度10%-100%    --->    SPI数据：1-10

/****************************************************************************
FM调制度频偏5kHz ---> SPI数据：1 	频偏10kHz ---> SPI数据：2

/****************************************************************************
调节VPP	10%-100%    --->    SPI数据：1-10

/****************************************************************************
			按键：
	上 -----> 频率步进+100	/	频偏10kHz
	下 -----> 频率步进-100	/	频偏5kHz
	右 -----> AM调制度步进+10%/	峰峰值调节步进+10%
	左 -----> AM调制度步进-10% /	峰峰值调节步进-10%
